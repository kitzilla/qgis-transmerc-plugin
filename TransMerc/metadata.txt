[general]
name=TransMerc
qgisMinimumVersion=1.7
description=This is a plugin for transforming the projections of vector layers to Transverse Mercator (TM) accurately and universally. The default projection engine of QGIS uses approximated Gauss Kruger Formulae for TM transformation, whose accuracy go worse or raise exceptions when data sit far away from the prime meridian (central longitude). Because this plugin uses the original formulae for transformation, it can perform TM transformation for any combination of geographical data and projection parameters. 
category=Vector
